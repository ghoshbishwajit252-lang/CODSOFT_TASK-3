""" P2P Rock-Paper-Scissors Game - Pure Python Run this script on both devices. One acts as server, other as client. """ import socket import threading import random import time import sys import json from datetime import datetime class RPSP2P: def __init__(self): self.host = self.get_local_ip() self.port = 12345 self.socket = None self.connection = None self.connected = False self.is_server = False self.opponent_address = None # Game state self.my_score = 0 self.opponent_score = 0 self.ties = 0 self.my_move = None self.opponent_move = None self.round_complete = False self.my_turn = False self.game_over = False # Colors for terminal self.GREEN = '\033[92m' self.RED = '\033[91m' self.YELLOW = '\033[93m' self.BLUE = '\033[94m' self.CYAN = '\033[96m' self.MAGENTA = '\033[95m' self.RESET = '\033[0m' self.BOLD = '\033[1m' self.move_emojis = { 'rock': '🪨', 'paper': '📄', 'scissors': '✂️' } def get_local_ip(self): """Get local IP address""" try: s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM) s.connect(('8.8.8.8', 80)) ip = s.getsockname()[0] s.close() return ip except: return '127.0.0.1' def clear_screen(self): """Clear terminal screen""" print('\033[2J\033[H', end='') def print_header(self): """Print game header""" self.clear_screen() print(f"{self.CYAN}{'='*60}{self.RESET}") print(f"{self.BOLD}{self.MAGENTA}🌐 P2P ROCK-PAPER-SCISSORS 🌐{self.RESET}") print(f"{self.CYAN}{'='*60}{self.RESET}") print(f"{self.BLUE}📡 Local IP: {self.host}:{self.port}{self.RESET}") if self.connected and self.opponent_address: print(f"{self.BLUE}👥 Connected to: {self.opponent_address}{self.RESET}") print(f"{self.BOLD}{self.GREEN}✅ CONNECTED{self.RESET}" if self.connected else f"{self.RED}❌ DISCONNECTED{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") def print_scoreboard(self): """Display current scores""" print(f"\n{self.BOLD}{'SCOREBOARD':^60}{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") print(f"{self.GREEN}🧑 You: {self.my_score}{self.RESET}") print(f"{self.RED}🤖 Opponent: {self.opponent_score}{self.RESET}") print(f"{self.YELLOW}🤝 Ties: {self.ties}{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") def print_moves(self): """Display current round moves""" print(f"\n{self.BOLD}{'ROUND MOVES':^60}{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") your_move = self.move_emojis.get(self.my_move, '❓') + ' ' + (self.my_move or '?') opp_move = self.move_emojis.get(self.opponent_move, '❓') + ' ' + (self.opponent_move or '?') if self.my_move and self.opponent_move: print(f"{self.GREEN}🧑 You: {your_move}{self.RESET}") print(f"{self.RED}🤖 Opponent: {opp_move}{self.RESET}") else: print(f"{self.GREEN}🧑 You: {your_move}{self.RESET}") if self.my_move and not self.opponent_move: print(f"{self.YELLOW}⏳ Waiting for opponent...{self.RESET}") else: print(f"{self.RED}🤖 Opponent: {opp_move}{self.RESET}") def print_result(self, result): """Display round result""" print(f"\n{self.BOLD}{'RESULT':^60}{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") if result == 'win': print(f"{self.GREEN}🎉 YOU WIN! {self.move_emojis[self.my_move]} beats {self.move_emojis[self.opponent_move]}{self.RESET}") elif result == 'lose': print(f"{self.RED}😔 YOU LOSE! {self.move_emojis[self.opponent_move]} beats {self.move_emojis[self.my_move]}{self.RESET}") else: print(f"{self.YELLOW}🤝 IT'S A TIE! Both chose {self.move_emojis[self.my_move]}{self.RESET}") def start_server(self): """Start as server (wait for connection)""" try: self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM) self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1) self.socket.bind((self.host, self.port)) self.socket.listen(1) self.is_server = True self.print_header() print(f"{self.YELLOW}📡 Waiting for connection on {self.host}:{self.port}...{self.RESET}") print(f"{self.YELLOW}💡 Tell your opponent to connect to: {self.host}{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") self.connection, addr = self.socket.accept() self.opponent_address = f"{addr[0]}:{addr[1]}" self.connected = True self.my_turn = True # Server goes first print(f"{self.GREEN}✅ Connected to {self.opponent_address}!{self.RESET}") print(f"{self.GREEN}🎯 You go first!{self.RESET}") # Start game loop self.game_loop() except Exception as e: print(f"{self.RED}❌ Server error: {e}{self.RESET}") input("\nPress Enter to exit...") def connect_to_server(self, server_ip): """Connect to a server""" try: self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM) self.socket.connect((server_ip, self.port)) self.connection = self.socket self.connected = True self.is_server = False self.opponent_address = f"{server_ip}:{self.port}" self.my_turn = False # Client waits for server print(f"{self.GREEN}✅ Connected to {self.opponent_address}!{self.RESET}") print(f"{self.YELLOW}⏳ Waiting for opponent to start...{self.RESET}") # Start game loop self.game_loop() except Exception as e: print(f"{self.RED}❌ Connection error: {e}{self.RESET}") input("\nPress Enter to exit...") def send_message(self, message_type, data=None): """Send a message to opponent""" try: msg = json.dumps({'type': message_type, 'data': data}) self.connection.send((msg + '\n').encode()) return True except: return False def receive_message(self): """Receive a message from opponent""" try: data = self.connection.recv(1024).decode() if data: return json.loads(data.strip()) except: pass return None def game_loop(self): """Main game loop with threading for receiving""" # Start receiving thread receive_thread = threading.Thread(target=self.receive_handler, daemon=True) receive_thread.start() # Game loop while self.connected and not self.game_over: self.print_header() self.print_scoreboard() self.print_moves() if self.round_complete: # Show result and wait for next round print(f"{self.CYAN}{'-'*60}{self.RESET}") print(f"{self.YELLOW}📝 Press Enter to continue to next round...{self.RESET}") input() self.reset_round() continue if self.my_turn and not self.round_complete: print(f"\n{self.GREEN}🎯 Your turn! Choose your move:{self.RESET}") print(f"{self.CYAN}{'─'*60}{self.RESET}") print(f"{self.BOLD} 1. Rock 🪨") print(f" 2. Paper 📄") print(f" 3. Scissors ✂️") print(f" 4. Quit 🚪") print(f"{self.CYAN}{'─'*60}{self.RESET}") choice = input(f"{self.BOLD}👉 Enter your choice (1-4): {self.RESET}").strip() if choice == '4': self.quit_game() break move_map = {'1': 'rock', '2': 'paper', '3': 'scissors'} if choice in move_map: self.my_move = move_map[choice] # Send move to opponent if self.send_message('move', self.my_move): self.my_turn = False print(f"{self.BLUE}📤 Move sent! Waiting for opponent...{self.RESET}") time.sleep(0.5) else: print(f"{self.RED}❌ Failed to send move{self.RESET}") else: print(f"{self.RED}❌ Invalid choice! Press Enter...{self.RESET}") input() else: # Waiting for opponent print(f"\n{self.YELLOW}⏳ Waiting for opponent's move...{self.RESET}") if not self.connected: break time.sleep(0.5) if not self.game_over: self.cleanup() def receive_handler(self): """Handle incoming messages in background thread""" while self.connected: msg = self.receive_message() if msg: if msg['type'] == 'move': self.opponent_move = msg['data'] # Check if both have moved if self.my_move and self.opponent_move: self.resolve_round() elif msg['type'] == 'reset': self.reset_round() elif msg['type'] == 'quit': self.quit_game() elif msg['type'] == 'handshake': # Reply to handshake self.send_message('handshake_ack', {'connected': True}) else: # Connection lost if self.connected: print(f"\n{self.RED}❌ Connection lost!{self.RESET}") self.connected = False self.game_over = True break def resolve_round(self): """Determine round winner""" if not self.my_move or not self.opponent_move: return # Determine result if self.my_move == self.opponent_move: result = 'tie' self.ties += 1 elif ((self.my_move == 'rock' and self.opponent_move == 'scissors') or (self.my_move == 'scissors' and self.opponent_move == 'paper') or (self.my_move == 'paper' and self.opponent_move == 'rock')): result = 'win' self.my_score += 1 else: result = 'lose' self.opponent_score += 1 self.round_complete = True # Send result to opponent (if they haven't already computed it) self.send_message('round_result', { 'my_move': self.my_move, 'opponent_move': self.opponent_move, 'result': result }) # Display result self.print_header() self.print_scoreboard() self.print_moves() self.print_result(result) # Update turn for next round (alternate) self.my_turn = not self.my_turn def reset_round(self): """Reset for next round""" self.my_move = None self.opponent_move = None self.round_complete = False # Keep the turn as it was set in resolve_round def quit_game(self): """Quit the game""" self.game_over = True self.send_message('quit') self.cleanup() def cleanup(self): """Close connections""" self.connected = False if self.connection: try: self.connection.close() except: pass if self.socket: try: self.socket.close() except: pass self.print_header() print(f"{self.YELLOW}👋 Game ended! Final scores:{self.RESET}") self.print_scoreboard() print(f"\n{self.GREEN}Thanks for playing!{self.RESET}") input("\nPress Enter to exit...") sys.exit(0) def main(): """Main entry point""" print(f"{self.CYAN}{'='*60}{self.RESET}") print(f"{self.BOLD}{self.MAGENTA}🌐 P2P ROCK-PAPER-SCISSORS 🌐{self.RESET}") print(f"{self.CYAN}{'='*60}{self.RESET}") print(f"\n{self.BOLD}Choose your role:{self.RESET}") print(f" 1. {self.GREEN}Start as Server{self.RESET} (Wait for opponent to connect)") print(f" 2. {self.BLUE}Connect to Server{self.RESET} (Join an existing game)") print(f" 3. {self.RED}Exit{self.RESET}") print(f"{self.CYAN}{'-'*60}{self.RESET}") choice = input(f"{self.BOLD}👉 Enter your choice (1-3): {self.RESET}").strip() game = RPSP2P() if choice == '1': game.start_server() elif choice == '2': server_ip = input(f"{self.BOLD}Enter server IP address: {self.RESET}").strip() if not server_ip: server_ip = game.get_local_ip() print(f"{self.YELLOW}💡 Using local IP: {server_ip}{self.RESET}") game.connect_to_server(server_ip) else: print(f"{self.YELLOW}👋 Goodbye!{self.RESET}") sys.exit(0) if __name__ == "__main__": main()
